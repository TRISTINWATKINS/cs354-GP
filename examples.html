Tristin Watkins — Medium Program

// Medium Rust Program: Guessing Thing (super basic)
// I only know how to read input and compare numbers

use std::io;

fn main() {
    println!("Guess the number (it's 7 btw lol):");

    let mut user_guess = String::new();

    io::stdin()
        .read_line(&mut user_guess)
        .expect("could not read");

    let guess_num: i32 = user_guess.trim().parse().unwrap_or(0);

    if guess_num == 7 {
        println!("You got it right!");
    } else {
        println!("Nope, that wasnt it.");
    }
}
